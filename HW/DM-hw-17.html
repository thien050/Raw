Evaluating neural networks

sum(sephora$partition == "Train")/30
## [1] 213
sephora %>%
  dplyr::select(YJ_love,partition) %>%
  dplyr::mutate(
    nnet_3 = predict(
      object = nnet_3,
      newdata = sephora %>%
        dplyr::select(dplyr::any_of(pred))
    
  )) %>%
  tidyr::gather(key = "model",value = "predict",-YJ_love,-partition) %>%
  dplyr::mutate(residual = predict - YJ_love) %>%
  dplyr::group_by(model,partition) %>%
  dplyr::summarise(rmse = sqrt(mean(residual^2))) %>%
  tidyr::spread(key = partition,value = rmse) %>%
  dplyr::mutate(difference = Train - Test) %>%
  dplyr::arrange(Test,Train)
